            var QueryRouter = ExoSuit.Router.extend({
                mixins: [ ExoSuit.Mixins.ParseQueryStringMixin ],

                routes: {
                    "": "home"
                },

                initialize: function() {
                    Backbone.history.start();
                },

                home: function( qs ) {
                    $( '#query' ).html( JSON.stringify( qs, null, 4 ) );
                }
            });

            var app = new QueryRouter();
