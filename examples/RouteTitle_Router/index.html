            var TitleRouter = ExoSuit.Router.extend({
                mixins: [ ExoSuit.Mixins.RouteTitleMixin ],

                routes: {
                    "": "home",
                    "user-list": "userList",
                    "user/:id": "user",
                    "admin": "admin"
                },

                routeConfig: {
                    home: {
                        title: "Homepage"
                    },
                    userList: {
                        title: "User List"
                    },
                    user: {
                        title: function( args ) {
                            return "User: " + args[ 0 ];
                        }
                    }
                },

                defaultTitle: "Default Title",
                titlePrefix: "My Site - ",

                initialize: function() {
                    Backbone.history.start();
                },

                home: function() {
                    console.log( "Home Route" );
                },
                userList: function() {
                    console.log( "User List Route" );
                },
                user: function() {
                    console.log( "User Route" );
                },
                admin: function() {
                    console.log( "Admin Route" );
                }
            });

            var app = new TitleRouter();
